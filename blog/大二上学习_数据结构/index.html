iterator的实现

从array和chain类中定义iterator里的typedef由来

#include <iostream>
#include <iterator>
#include <vector>

template<typename T>
class MyBidirectionalIterator {
public:
    using iterator_category = std::bidirectional_iterator_tag; // 迭代器类别
    using value_type = T;
    using difference_type = std::ptrdiff_t;
    using pointer = T*;
    using reference = T&;
    MyBidirectionalIterator(pointer ptr) : _ptr(ptr) {}
    reference operator*() const {
        return *_ptr;
    }
    // 前缀递增
    MyBidirectionalIterator& operator++() {
        ++_ptr;
        return *this;
    }
    // 后缀递增
    MyBidirectionalIterator operator++(int) {
        MyBidirectionalIterator tmp = *this;
        ++_ptr;
        return tmp;
    }
    // 前缀递减
    MyBidirectionalIterator& operator--() {
        --_ptr;
        return *this;
    }
    // 后缀递减
    MyBidirectionalIterator operator--(int) {
        MyBidirectionalIterator tmp = *this;
        --_ptr;
        return tmp;
    }
    // 比较运算符
    bool operator==(const MyBidirectionalIterator& other) const {
        return _ptr == other._ptr;
    }
    bool operator!=(const MyBidirectionalIterator& other) const {
        return _ptr != other._ptr;
    }
private:
    pointer _ptr; // 指向元素的指针
};
int main() {
    std::vector<int> vec = {1, 2, 3, 4, 5};
    MyBidirectionalIterator<int> it(vec.data());
    // 正向遍历
    for (size_t i = 0; i < vec.size(); ++i, ++it) {
        std::cout << *it << " "; // 输出: 1 2 3 4 5
    }
    std::cout << std::endl;
    // 反向遍历
    for (size_t i = 0; i < vec.size(); --it) {
        std::cout << *it << " "; // 输出: 5 4 3 2 1
    }
    return 0;
}
